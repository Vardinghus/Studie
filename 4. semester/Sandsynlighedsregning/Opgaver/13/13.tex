\textbf{(b)} Lav en fordeling, som har middelværdi 1.\\\
\textbf{(c)} Brug den anden definition af $G_X(s)$
\begin{equation}
G_X(s)=\sum_{k=0}^{\infty}s^kp_X(k)
\end{equation}

\begin{equation}
G_X(s)=\sum_{k=0}^2s^kp_X(k)=\frac{1}{2}s+\frac{1}{2}s^2
\end{equation}
